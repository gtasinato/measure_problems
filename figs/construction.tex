\begin{document}

\begin{tikzpicture}[tdplot_main_coords, scale = 2.5]

\coordinate (P) at ({1/sqrt(2)},{1/sqrt(2)},{0});

\coordinate (L0) at ({sqrt(3)/2},{0.5},{0});
\coordinate (L1) at ({0.5},{sqrt(3)/2},{0});
\coordinate (W) at ({1/sqrt(3)}, {-1/sqrt(3)}, {1/sqrt(3)});


% Draw shaded circle
\shade[ball color = lightgray,
    opacity = 0.5
] (0,0,0) circle (1cm);

% draw arcs
\tdplotsetrotatedcoords{0}{0}{0};
\draw[dashed,
    tdplot_rotated_coords,
] (0,0,0) circle (1);

    tdplot_rotated_coords,
] (0,0,0) circle (1);

% Axes in 3 d coordinate system

\draw[dashed, gray] (0,0,0) -- (0,0,1.30);
\draw[dashed, gray] (0,0,0) -- (0, 0, -1);
\draw[dashed, gray] ({1/sqrt(2)},{-1/sqrt(2)},{0}) -- ({-1/sqrt(2)},{1/sqrt(2)},{0});;


% Line from the origin to (P)
\draw[black,  -stealth] (0,0,0) -- (P);

% Add small circle at (P)
\draw[black, fill] (P) circle (0.5pt)

		node[above] {$\vec{l_1}$};

		node[below right] {$w$};

		\tdplotsetrotatedcoords{-60}{90}{0};
		\draw[
		tdplot_rotated_coords,
		blue
		] (1,0,0) arc (0:180:1);
		\draw[fill,
		tdplot_rotated_coords,
		red
		] ({-1/2}, {sqrt(3)/2}, {0}) circle (0.5pt)
		node[above right, red] {$w'$};


\end{tikzpicture}
